/** * 學生會意見平台 - 後端核心 (GAS) * 功能：處理資料讀取(GET)與寫入/刪除/設定(POST) */ function doGet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. 抓取回饋資料 (工作表: data) var feedbackSheet = getOrCreateSheet(ss, "data", ["ID", "Category", "Text", "Likes", "Time"]); var fRows = feedbackSheet.getDataRange().getValues(); var feedbacks = []; for (var i = 1; i < fRows.length; i++) { feedbacks.push({ id: fRows[i][0], category: fRows[i][1], text: fRows[i][2], likes: fRows[i][3], time: fRows[i][4] }); } // 2. 抓取公告資料 (工作表: notices) var noticeSheet = getOrCreateSheet(ss, "notices", ["ID", "Content", "Time"]); var notices = []; var nRows = noticeSheet.getDataRange().getValues(); for (var j = nRows.length - 1; j >= 1; j--) { notices.push({ id: nRows[j][0], content: nRows[j][1], time: nRows[j][2] }); } // 3. 抓取配置資料 (工作表: config) var configSheet = getOrCreateSheet(ss, "config", ["Title", "Categories", "BgColor"]); var cRows = configSheet.getDataRange().getValues(); var config = { title: cRows[1] ? cRows[1][0] : "學生會意見平台", categories: cRows[1] ? cRows[1][1] : "生活設施,學業課程,社團活動,其他", bgColor: cRows[1] ? cRows[1][2] : "#f5f7f9", notices: notices }; return ContentService.createTextOutput(JSON.stringify({ feedbacks: feedbacks, config: config })).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var params = JSON.parse(e.postData.contents); var result = { result: "success" }; // --- 處理：學生提交意見 --- if (params.action === "addFeedback") { var sheet = ss.getSheetByName("data"); sheet.appendRow([Date.now(), params.category, params.text, 0, new Date().toLocaleString()]); } // --- 處理：管理員發布公告 --- else if (params.action === "addNotice") { var noticeSheet = ss.getSheetByName("notices"); noticeSheet.appendRow([Date.now(), params.content, new Date().toLocaleString()]); } // --- 處理：按讚 --- else if (params.action === "likeFeedback") { var sheet = ss.getSheetByName("data"); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] == params.id) { sheet.getRange(i + 1, 4).setValue(Number(rows[i][3]) + 1); break; } } } // --- 處理：刪除公告 --- else if (params.action === "deleteNotice") { var noticeSheet = ss.getSheetByName("notices"); var rows = noticeSheet.getDataRange().getValues(); if (params.type === 'latest') { noticeSheet.deleteRow(rows.length); } else { for (var i = 1; i < rows.length; i++) { if (rows[i][0] == params.id) { noticeSheet.deleteRow(i + 1); break; } } } } // --- 處理：刪除學生回覆 (修復同步失敗的核心) --- else if (params.action === "deleteFeedback") { var sheet = ss.getSheetByName("data"); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] == params.id) { sheet.deleteRow(i + 1); break; } } } // --- 處理：更新配置 (包含背景顏色) --- else if (params.action === "updateConfig") { var configSheet = ss.getSheetByName("config"); configSheet.getRange(2, 1, 1, 3).setValues([[params.title, params.categories, params.bgColor]]); } return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } // 輔助函式：確保工作表存在 function getOrCreateSheet(ss, name, headers) { var sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); } return sheet; }
